void
exp_op(_rr,_aa , _bb)
int *_rr,_aa,_bb;
{
  int ii,kk;
  ii = _bb;
  kk = 1;
  while ( ii != 0 ) {
    kk = kk*_aa;
    ii = ii-1;
  };
  *_rr = kk;
}
