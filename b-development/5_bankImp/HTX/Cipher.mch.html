MACHINE	Cipher(ee, dd, nn)

CONSTRAINTS
	ee : NAT &
	dd : NAT &
	nn : NAT1 &
	ee*dd mod nn-1 = 1

CONSTANTS 
	encrypt, decrypt


PROPERTIES
	encrypt : NAT --> NAT &
	decrypt : NAT --> NAT &
	
	!mm.(mm : 0 .. nn-1 => encrypt(decrypt(mm)) = mm) &
	!mm.(mm : 0 .. nn-1 => decrypt(encrypt(mm)) = mm) 

	/*Fermat's theorem*/
/*	
	ee * dd mod fi(nn) = 1 &
	gcd(ee, nn) = 1 &
*/

OPERATIONS
	
	rr <-- encrypt_op(mm) =
		PRE
			mm : 0 .. nn-1 
		THEN
			rr := encrypt(mm)
		END;
		
	rr <-- decrypt_op(mm) =
		PRE
			mm : 0 .. nn-1
		THEN
			rr := decrypt(mm)
		END
END
